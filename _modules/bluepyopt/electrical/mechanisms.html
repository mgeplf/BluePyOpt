[docs]    def instantiate(self, cell):
        """Instantiate"""

        for location in self.locations:
            isec_list = location.instantiate(cell)
            for isec in isec_list:
                isec.insert(self.prefix)
            logger.debug('Inserted %s in %s',
                         self.prefix, self.locations)
